    Universal launcher for Claude/Codex/Copilot CLI sessions and tasks, with FancyZones-style window tiling.

    .\launch.ps1 task TASK_k3s-keda

    .\launch.ps1 task -Next -Tool copilot

# ── Build zone grid (2x2 for landscape, 3x1 for portrait) ─────────────
$spacing = 5
$zones = @()
$monitorIndex = 0

$monitors = [System.Windows.Forms.Screen]::AllScreens |
    Sort-Object { $_.WorkingArea.X }

foreach ($mon in $monitors) {
    $monitorIndex++
    $wa = $mon.WorkingArea
    $isPortrait = $wa.Height -gt $wa.Width

    if ($isPortrait) {
        # Portrait: 3 rows, full width (3x1)
        $thirdH = [math]::Floor(($wa.Height - 2 * $spacing) / 3)
        $fullW  = $wa.Width

        $y0 = $wa.Y
        $y1 = $wa.Y + $thirdH + $spacing
        $y2 = $wa.Y + 2 * ($thirdH + $spacing)

        $zones += [PSCustomObject]@{ Index=$zones.Count; Monitor=$monitorIndex; X=$wa.X; Y=$y0; W=$fullW; H=$thirdH }
        $zones += [PSCustomObject]@{ Index=$zones.Count; Monitor=$monitorIndex; X=$wa.X; Y=$y1; W=$fullW; H=$thirdH }
        $zones += [PSCustomObject]@{ Index=$zones.Count; Monitor=$monitorIndex; X=$wa.X; Y=$y2; W=$fullW; H=$thirdH }
    } else {
        # Landscape: 2x2 grid
        $halfW = [math]::Floor(($wa.Width  - $spacing) / 2)
        $halfH = [math]::Floor(($wa.Height - $spacing) / 2)

        $x0 = $wa.X;  $x1 = $wa.X + $halfW + $spacing
        $y0 = $wa.Y;  $y1 = $wa.Y + $halfH + $spacing

        $zones += [PSCustomObject]@{ Index=$zones.Count; Monitor=$monitorIndex; X=$x0; Y=$y0; W=$halfW; H=$halfH }
        $zones += [PSCustomObject]@{ Index=$zones.Count; Monitor=$monitorIndex; X=$x1; Y=$y0; W=$halfW; H=$halfH }
        $zones += [PSCustomObject]@{ Index=$zones.Count; Monitor=$monitorIndex; X=$x0; Y=$y1; W=$halfW; H=$halfH }
        $zones += [PSCustomObject]@{ Index=$zones.Count; Monitor=$monitorIndex; X=$x1; Y=$y1; W=$halfW; H=$halfH }
    }
}

# ── Zone management ───────────────────────────────────────────────────
function Get-OccupiedZones {
    $occupied = @()
    $infoFiles = Get-ChildItem -Path $progressDir -Filter '*.launch-info.json' -ErrorAction SilentlyContinue
    foreach ($f in $infoFiles) {
        try {
            $info = Get-Content $f.FullName -Raw | ConvertFrom-Json
            # Only count as occupied if the process is still running
            $proc = Get-Process -Id $info.PID -ErrorAction SilentlyContinue
            if ($proc) {
                $occupied += $info.Zone
            } else {
                # Stale info file -- process ended
                Remove-Item $f.FullName -Force -ErrorAction SilentlyContinue
            }
        } catch {
            Remove-Item $f.FullName -Force -ErrorAction SilentlyContinue
        }
    }
    return $occupied
}

function Get-NextEmptyZone {
    param([int]$MonitorFilter = 0)
    $occupied = Get-OccupiedZones
    foreach ($z in $zones) {
        if ($z.Index -notin $occupied) {
            if ($MonitorFilter -eq 0 -or $z.Monitor -eq $MonitorFilter) { return $z }
        }
    }
    return $null
}

function Set-WindowZone {
    param([int]$ProcessId, [PSCustomObject]$Zone)
    # Poll for window handle -- Claude can take a few seconds to create its window
    $hwnd = [IntPtr]::Zero
    for ($i = 0; $i -lt 10; $i++) {
        Start-Sleep -Milliseconds 500
        $proc = Get-Process -Id $ProcessId -ErrorAction SilentlyContinue
        if ($proc -and $proc.MainWindowHandle -ne [IntPtr]::Zero) {
            $hwnd = $proc.MainWindowHandle
            break
        }
    }
    if ($hwnd -ne [IntPtr]::Zero) {
        [Win32Launch]::ShowWindow($hwnd, [Win32Launch]::SW_RESTORE) | Out-Null
        [Win32Launch]::SetWindowPos($hwnd, [Win32Launch]::HWND_TOP,
            $Zone.X, $Zone.Y, $Zone.W, $Zone.H,
            [Win32Launch]::SWP_SHOWWINDOW) | Out-Null
        Write-Host "  Tiled to zone $($Zone.Index) [$($Zone.X),$($Zone.Y) $($Zone.W)x$($Zone.H)]" -ForegroundColor DarkGray
    } else {
        Write-Warning "  Could not get window handle for PID $ProcessId - tile manually"
    }
}

# ── Move task + companion folder helper ────────────────────────────────
function Move-TaskItem {
    param(
        [string]$FilePath,
        [string]$DestDir
    )
    $file = Get-Item $FilePath
    $slug = $file.BaseName
    $srcDir = $file.DirectoryName

    # Move the .md file
    $destFile = Join-Path $DestDir $file.Name
    Move-Item $file.FullName -Destination $destFile -Force
    Write-Host "  Moved $($file.Name) -> $(Split-Path $DestDir -Leaf)/" -ForegroundColor Cyan

    # Move companion folder (same name as slug) if it exists
    $companionDir = Join-Path $srcDir $slug
    if (Test-Path $companionDir -PathType Container) {
        $destCompanion = Join-Path $DestDir $slug
        Move-Item -Path $companionDir -Destination $destCompanion -Force
        Write-Host "  Moved $slug/ -> $(Split-Path $DestDir -Leaf)/$slug/" -ForegroundColor Cyan
    }

    return $destFile
}

# ── Resolve task files ────────────────────────────────────────────────
function Resolve-TaskFiles {
    param(
        [string]$Task,
        [switch]$NextMode,
        [int]$Count
    )

    $taskFiles = @()

    if ($Task) {
        $matches = Get-ChildItem -Path $backlogDir -Filter "*$Task*.md" -ErrorAction SilentlyContinue
        if (-not $matches -or $matches.Count -eq 0) {
            # Also check in-progress in case it was already moved
            $matches = Get-ChildItem -Path $progressDir -Filter "*$Task*.md" -ErrorAction SilentlyContinue |
                Where-Object { $_.Name -notlike '*.launch-info.json' }
            if ($matches -and $matches.Count -gt 0) {
                Write-Host "Task '$Task' is already in-progress." -ForegroundColor Yellow
                $taskFiles += $matches | Select-Object -First 1
            } else {
                Write-Error "No task matching '*$Task*.md' found in backlog or in-progress."
                return @()
            }
        } else {
            $taskFiles += $matches | Select-Object -First $Count
        }
    } elseif ($NextMode) {
        $available = Get-ChildItem -Path $backlogDir -Filter '*.md' -ErrorAction SilentlyContinue |
            Where-Object { $_.Name -ne 'README.md' } |
            Sort-Object Name
        if (-not $available -or $available.Count -eq 0) {
            Write-Host 'No tasks in backlog.' -ForegroundColor Yellow
            return @()
        }
        $taskFiles = $available | Select-Object -First $Count
    }

    return $taskFiles
}

# ── Inject text into a target console (focus-free) ───────────────────
# Uses WriteConsoleInput via a hidden helper process: FreeConsole from its own
# console, AttachConsole to target, write key events, detach. No focus needed.
function Send-ConsoleInput {
    param(
        [int]$TargetPid,
        [string]$Text
    )

    $script = @"
`$ErrorActionPreference = 'Stop'
try {
    Add-Type -TypeDefinition @'
using System;
using System.Runtime.InteropServices;
public class ConInject {
    [DllImport("kernel32.dll")] public static extern bool FreeConsole();
    [DllImport("kernel32.dll")] public static extern bool AttachConsole(uint pid);
    [DllImport("kernel32.dll")] public static extern IntPtr GetStdHandle(int h);
    [StructLayout(LayoutKind.Explicit, Size = 20)]
    public struct INPUT_RECORD {
        [FieldOffset(0)]  public ushort EventType;
        [FieldOffset(4)]  public int    bKeyDown;
        [FieldOffset(8)]  public ushort wRepeatCount;
        [FieldOffset(10)] public ushort wVirtualKeyCode;
        [FieldOffset(12)] public ushort wVirtualScanCode;
        [FieldOffset(14)] public char   UnicodeChar;
        [FieldOffset(16)] public uint   dwControlKeyState;
    }
    [DllImport("kernel32.dll", SetLastError = true)]
    public static extern bool WriteConsoleInput(
        IntPtr h, INPUT_RECORD[] buf, uint len, out uint written);
    public static bool Send(uint pid, string text) {
        FreeConsole();
        if (!AttachConsole(pid)) return false;
        IntPtr h = GetStdHandle(-10);
        string full = text + "\r";
        var r = new INPUT_RECORD[full.Length * 2];
        for (int i = 0; i < full.Length; i++) {
            char c = full[i];
            ushort vk = (c == '\r') ? (ushort)0x0D : (ushort)0;
            r[i*2].EventType = 1; r[i*2].bKeyDown = 1;
            r[i*2].wRepeatCount = 1; r[i*2].wVirtualKeyCode = vk;
            r[i*2].UnicodeChar = c;
            r[i*2+1].EventType = 1; r[i*2+1].bKeyDown = 0;
            r[i*2+1].wRepeatCount = 1; r[i*2+1].wVirtualKeyCode = vk;
            r[i*2+1].UnicodeChar = c;
        }
        uint w;
        return WriteConsoleInput(h, r, (uint)r.Length, out w);
    }
}
'@
    if ([ConInject]::Send(~~PID~~, '~~TEXT~~')) { exit 0 } else { exit 1 }
} catch { exit 2 }
"@

    $script = $script.Replace('~~PID~~', $TargetPid.ToString()).Replace('~~TEXT~~', $Text.Replace("'", "''"))
    $encoded = [Convert]::ToBase64String([System.Text.Encoding]::Unicode.GetBytes($script))
    $p = Start-Process powershell -ArgumentList '-NoProfile', '-WindowStyle', 'Hidden', '-EncodedCommand', $encoded -Wait -PassThru
    return ($p.ExitCode -eq 0)
}

# ── Set input bar name + color via WriteConsoleInput ─────────────────
# The input box label reads standaloneAgentContext.name, which is ONLY set
# by the /rename command in-process. --name sets currentSessionTitle (terminal
# title / resume picker) but not the input bar. JSONL injection can't work
# because the file watcher doesn't update in-memory state for custom-title.
#
# Strategy: launch Claude WITH the initial prompt, then inject /color + /rename
# via WriteConsoleInput into the console's input buffer (no focus needed).
# Both commands are immediate:true, so they work even when the assistant is
# mid-turn. We wait for the JSONL to have content (proves the TUI is live
# and processing), then 3s settle before injecting.
function Set-InputBarName {
    param(
        [string]$DisplayName,
        [string]$Color,
        [int]$ProcessId,
        [string]$WorkDir,
        [datetime]$LaunchedAfter,
        [string]$WindowTitle = ''   # cmd.exe title for AppActivate fallback
    )

    $sessionsDir = Join-Path $env:USERPROFILE '.claude\sessions'

    # 1. Find the Claude child process PID inside cmd.exe, then read its session file.
    #    Session files are named <pid>.json — matching by PID is precise (cwd matches
    #    every session in $HOME, which caused false matches).
    $sessionId = $null
    $claudePid = $null

    Write-Host '  Resolving Claude PID...' -ForegroundColor DarkGray
    for ($i = 0; $i -lt 60; $i++) {   # 30 seconds to find child
        Start-Sleep -Milliseconds 500
        # Walk descendant tree: cmd.exe -> node.exe/claude.exe (may be nested)
        $descendants = @()
        $frontier = @($ProcessId)
        for ($depth = 0; $depth -lt 3; $depth++) {
            $nextFrontier = @()
            foreach ($p in $frontier) {
                $children = Get-CimInstance Win32_Process -Filter "ParentProcessId = $p" -ErrorAction SilentlyContinue
                foreach ($c in $children) {
                    $descendants += $c.ProcessId
                    $nextFrontier += $c.ProcessId
                }
            }
            $frontier = $nextFrontier
        }
        # Check each descendant for a session file
        foreach ($dp in $descendants) {
            $sf = Join-Path $sessionsDir "$dp.json"
            if (Test-Path $sf) {
                try {
                    $pidData = Get-Content $sf -Raw | ConvertFrom-Json
                    if ($pidData.sessionId) {
                        $claudePid = $dp
                        $sessionId = $pidData.sessionId
                        Write-Host "  Claude PID: $claudePid  Session: $sessionId" -ForegroundColor DarkGray
                        break
                    }
                } catch {}
            }
        }
        if ($sessionId) { break }
    }

    if (-not $sessionId) {
        Write-Warning '  Could not find Claude session (no child with session file after 30s)'
        return $null
    }

    # 2. Wait for TUI to be actively processing — poll for JSONL to exist and grow.
    #    The JSONL is written when the session starts processing (custom-title, then
    #    the first user message + assistant response). Its existence proves the TUI
    #    is live and has started handling the initial prompt.
    $encodedDir = $WorkDir -replace ':', '-' -replace '\\', '-' -replace '/', '-'
    $jsonlPath = Join-Path $env:USERPROFILE ".claude\projects\$encodedDir\$sessionId.jsonl"

    Write-Host '  Waiting for TUI activity (JSONL)...' -ForegroundColor DarkGray
    $tuiReady = $false
    for ($j = 0; $j -lt 120; $j++) {   # 60 seconds max
        Start-Sleep -Milliseconds 500
        if (Test-Path $jsonlPath) {
            $sz = (Get-Item $jsonlPath -ErrorAction SilentlyContinue).Length
            if ($sz -gt 200) { $tuiReady = $true; break }
        }
    }

    if (-not $tuiReady) {
        Write-Warning '  JSONL not ready -- TUI may not be initialized. Sending anyway.'
    } else {
        Write-Host ('  TUI active (JSONL ' + $sz + ' bytes)') -ForegroundColor DarkGray
    }

    # Settle time for TUI render loop
    Start-Sleep -Seconds 3

    # 3. Inject /color + /rename via WriteConsoleInput (no focus needed).
    #    Both commands are immediate:true — work even while assistant is mid-processing.
    if ($Color) {
        $ok = Send-ConsoleInput -TargetPid $ProcessId -Text ('/color ' + $Color)
        Write-Host ('  /color ' + $Color + ': ' + $(if ($ok) {'injected'} else {'FAILED'})) -ForegroundColor DarkGray
        Start-Sleep -Milliseconds 800
    }

    $ok = Send-ConsoleInput -TargetPid $ProcessId -Text ('/rename ' + $DisplayName)
    Write-Host ('  /rename ' + $DisplayName + ': ' + $(if ($ok) {'injected'} else {'FAILED'})) -ForegroundColor DarkGray

    Write-Host ('  Input bar: ' + $DisplayName + ' (' + $Color + ')') -ForegroundColor Cyan

    return $sessionId
}

# ── Launch a Claude session for a task ─────────────────────────────────
function Launch-TaskSession {
    param(
        [System.IO.FileInfo]$TaskFile,
        [switch]$Resume,
        [switch]$SkipPerms,
        [switch]$Remote,
        [string]$RemoteHost = 'ubuntu1'
    )

    $taskSlug = $TaskFile.BaseName

    # Move to in-progress (if still in backlog)
    if ($TaskFile.DirectoryName -eq (Resolve-Path $backlogDir).Path) {
        $destPath = Move-TaskItem -FilePath $TaskFile.FullName -DestDir $progressDir
    } else {
        $destPath = $TaskFile.FullName
    }

    # Brief pause so the file is settled
    Start-Sleep -Seconds 1

    # Read task content for context prompt
    $taskContent = Get-Content $destPath -Raw -ErrorAction SilentlyContinue
    $contextPrompt = "Picking up task: $taskSlug`n`nTask brief:`n$taskContent"

    # Resolve or create session ID
    $sessionId = $null
    $sessionStateRoot = Join-Path $env:USERPROFILE '.claude\projects'

    if ($Resume) {
        # Check for previous launch-info
        $prevInfo = Join-Path $progressDir "$taskSlug.launch-info.json"
        if (Test-Path $prevInfo) {
            try {
                $prev = Get-Content $prevInfo -Raw | ConvertFrom-Json
                if ($prev.SessionId -and $prev.SessionId -ne '') { $sessionId = $prev.SessionId }
            } catch {}
        }

        # If not found, search session transcripts for task slug
        if (-not $sessionId) {
            $jsonlFiles = Get-ChildItem $sessionStateRoot -Recurse -Filter '*.jsonl' -ErrorAction SilentlyContinue |
                Sort-Object LastWriteTime -Descending |
                Select-Object -First 20
            foreach ($jf in $jsonlFiles) {
                $sample = Get-Content $jf.FullName -TotalCount 50 -ErrorAction SilentlyContinue
                $joined = $sample -join ' '
                if ($joined -match [regex]::Escape($taskSlug)) {
                    $sessionId = $jf.BaseName
                    Write-Host "  Found previous session: $sessionId" -ForegroundColor Green
                    break
                }
            }
        }
    }

    # Write a batch launcher script (avoids all cmd.exe quoting issues)
    # Get a zone
    $zone = Get-NextEmptyZone -MonitorFilter $Monitor
    if (-not $zone) {
        Write-Warning "No empty zones available. Launching without tiling."
    }

    $cfg = $ToolConfig[$Tool]

    # ── Remote launch (SSH to VM) ────────────────────────────────────
    if ($Remote) {
        $remoteWorkDir = "/home/user/task-runs/$taskSlug"
        $remoteTaskPath = "/mnt/hgfs/tasks/in-progress/$($TaskFile.Name)"
        $remoteInitMsg = "Read the task file at $remoteTaskPath and begin working on it."

        # Remote tool config: Linux executable names + skip-perms flags
        $RemoteToolConfig = @{
            'claude'  = @{ Exe = 'claude';  SkipFlag = '--dangerously-skip-permissions'; PromptStyle = 'positional' }
            'codex'   = @{ Exe = 'codex';   SkipFlag = '--full-auto';                   PromptStyle = 'positional' }
            'copilot' = @{ Exe = 'copilot'; SkipFlag = '--allow-all';                   PromptStyle = 'interactive-flag' }
        }
        $rcfg = $RemoteToolConfig[$Tool]

        # Build prompt arg in tool's style
        switch ($rcfg.PromptStyle) {
            'positional'       { $remotePromptArg = " `"$remoteInitMsg`"" }
            'interactive-flag' { $remotePromptArg = " -i `"$remoteInitMsg`"" }
        }

        # Remote always skips permissions
        $remoteCmd = "$($rcfg.Exe) $($rcfg.SkipFlag)$remotePromptArg"
        $sshCmd = "ssh -t $RemoteHost `"mkdir -p $remoteWorkDir && cd $remoteWorkDir && $remoteCmd`""
        $toolLabel = $cfg.Label
        $cmdLine = 'title @' + $RemoteHost.ToUpper() + ' ' + $taskSlug + ' && ' + $sshCmd

        $launchTime = Get-Date
        $process = Start-Process cmd -ArgumentList '/k', $cmdLine -WorkingDirectory $env:USERPROFILE -PassThru

        # Claim the zone immediately so parallel launches see it as occupied
        $launchInfo = @{
            TaskSlug   = $taskSlug
            Tool       = $Tool
            Remote     = $RemoteHost
            PID        = $process.Id
            Zone       = if ($zone) { $zone.Index } else { -1 }
            SessionId  = ''
            LaunchedAt = (Get-Date -Format 'yyyy-MM-dd HH:mm:ss')
        } | ConvertTo-Json

        $infoPath = Join-Path $progressDir "$taskSlug.launch-info.json"
        Set-Content -Path $infoPath -Value $launchInfo -Encoding UTF8

        if ($zone -and $process) {
            Set-WindowZone -ProcessId $process.Id -Zone $zone
        }

        # Inject /color + /rename for Claude sessions (slash commands are Claude-specific)
        $displayName = "@$($RemoteHost.ToUpper()) " + ($taskSlug -replace '^TASK_', '')
        if ($Tool -eq 'claude') {
            Write-Host "  Waiting for remote TUI to initialize..." -ForegroundColor DarkGray
            Start-Sleep -Seconds 12

            $ok = Send-ConsoleInput -TargetPid $process.Id -Text '/color orange'
            Write-Host ('  /color yellow: ' + $(if ($ok) {'injected'} else {'FAILED'})) -ForegroundColor DarkGray
            Start-Sleep -Milliseconds 800

            $ok = Send-ConsoleInput -TargetPid $process.Id -Text ('/rename ' + $displayName)
            Write-Host ('  /rename ' + $displayName + ': ' + $(if ($ok) {'injected'} else {'FAILED'})) -ForegroundColor DarkGray
        }

        Write-Host "Launched REMOTE $toolLabel on ${RemoteHost}: $taskSlug (PID $($process.Id))" -ForegroundColor Green
        return
    }

    # ── Local launch ─────────────────────────────────────────────────
    $workDir = $env:USERPROFILE
    $initMsg = 'Read the task file at ' + $destPath + ' and begin working on it.'
    $skipFlag = ''
    if ($SkipPerms) { $skipFlag = ' ' + $cfg.SkipPermsFlag }

    # Write a temp .bat to avoid cmd.exe compound-command path issues
    $batFile = Join-Path $env:TEMP "launch-$taskSlug.bat"
    $batLines = @(
        '@echo off'
        'cd /d "' + $workDir + '"'
        'title ' + $cfg.Label + ': ' + $taskSlug
    )

    if ($sessionId) {
        switch ($cfg.ResumeStyle) {
            'flag'       { $batLines += '"' + $cfg.Exe + '" --resume ' + $sessionId }
            'subcommand' { $batLines += '"' + $cfg.Exe + '" resume ' + $sessionId }
            'flag-eq'    { $batLines += '"' + $cfg.Exe + '" --resume=' + $sessionId }
        }
    } else {
        switch ($cfg.PromptStyle) {
            'positional'       { $promptPart = ' "' + $initMsg + '"' }
            'interactive-flag' { $promptPart = ' -i "' + $initMsg + '"' }
        }
        $namePart = if ($cfg.SupportsName) { ' --name "' + $taskSlug + '"' } else { '' }
        $batLines += '"' + $cfg.Exe + '"' + $skipFlag + $namePart + $promptPart
    }

    Set-Content -Path $batFile -Value ($batLines -join "`r`n") -Encoding ASCII
    $cmdLine = $batFile

    $launchTime = Get-Date
    $process = Start-Process cmd -ArgumentList '/k', $cmdLine -WorkingDirectory $workDir -PassThru

    # Claim the zone immediately so parallel launches see it as occupied
    $infoPath = Join-Path $progressDir "$taskSlug.launch-info.json"
    $launchInfo = @{
        TaskSlug  = $taskSlug
        Tool      = $Tool
        PID       = $process.Id
        Zone      = if ($zone) { $zone.Index } else { -1 }
        SessionId = if ($sessionId) { $sessionId } else { '' }
        LaunchedAt = (Get-Date -Format 'yyyy-MM-dd HH:mm:ss')
    } | ConvertTo-Json
    Set-Content -Path $infoPath -Value $launchInfo -Encoding UTF8

    # Tile the window
    if ($zone -and $process) {
        Set-WindowZone -ProcessId $process.Id -Zone $zone
    }

    # Send /rename + /color to set input bar label and project color (Claude only)
    if (-not $sessionId -and $process -and $cfg.SupportsRename) {
        $taskColor = Get-TaskColor $taskSlug
        # Strip TASK_ prefix for a cleaner display name (slug only)
        $displayName = $taskSlug -replace '^TASK_', ''
        $resolvedId = Set-InputBarName -DisplayName $displayName -Color $taskColor -ProcessId $process.Id -WorkDir $workDir -LaunchedAfter $launchTime
        if ($resolvedId) { $sessionId = $resolvedId }

        # Update launch info with resolved session ID
        $launchInfo = @{
            TaskSlug  = $taskSlug
            Tool      = $Tool
            PID       = $process.Id
            Zone      = if ($zone) { $zone.Index } else { -1 }
            SessionId = if ($sessionId) { $sessionId } else { '' }
            LaunchedAt = (Get-Date -Format 'yyyy-MM-dd HH:mm:ss')
        } | ConvertTo-Json
        Set-Content -Path $infoPath -Value $launchInfo -Encoding UTF8
    }

    Write-Host "Launched: $taskSlug (PID $($process.Id))" -ForegroundColor Green
}

# ══════════════════════════════════════════════════════════════════════
# ── Main switch ──────────────────────────────────────────────────────
# ══════════════════════════════════════════════════════════════════════

switch ($Action.ToLower()) {

    "task" {
        $taskFiles = Resolve-TaskFiles -Task $Arg1 -NextMode:$Next -Count $Count
        if (-not $taskFiles -or $taskFiles.Count -eq 0) { return }

        foreach ($tf in $taskFiles) {
            Launch-TaskSession -TaskFile $tf -Resume:$Resume -SkipPerms:$SkipPerms -Remote:$Remote -RemoteHost $RemoteHost
        }
    }

    "status" {
        Write-Host "`nZone Status:" -ForegroundColor White
        Write-Host ("=" * 50)
        $infoFiles = Get-ChildItem -Path $progressDir -Filter '*.launch-info.json' -ErrorAction SilentlyContinue
        if (-not $infoFiles -or $infoFiles.Count -eq 0) {
            Write-Host "  No active tasks." -ForegroundColor DarkGray
        } else {
            foreach ($f in $infoFiles) {
                try {
                    $info = Get-Content $f.FullName -Raw | ConvertFrom-Json
                    $proc = Get-Process -Id $info.PID -ErrorAction SilentlyContinue
                    $alive = if ($proc) { "RUNNING" } else { "ENDED" }
                    $color = if ($proc) { "Green" } else { "DarkGray" }
                    $monNum = ($zones | Where-Object { $_.Index -eq $info.Zone }).Monitor
                    $toolLabel = if ($info.Tool) { $info.Tool } else { 'claude' }
                    Write-Host "  Mon${monNum} Zone $($info.Zone): $($info.TaskSlug) [$toolLabel] (PID $($info.PID)) [$alive]" -ForegroundColor $color
                } catch {
                    Write-Host "  (corrupt: $($f.Name))" -ForegroundColor Red
                }
            }
        }
        Write-Host "`nAvailable zones: $(($zones.Count) - (Get-OccupiedZones).Count) / $($zones.Count)" -ForegroundColor Cyan
        Write-Host ""
    }

    "done" {
        # Move a task from in-progress to done
        if (-not $Arg1) { Write-Error "Usage: launch.ps1 done <task-slug>"; return }
        $matches = Get-ChildItem -Path $progressDir -Filter "*$Arg1*.md" -ErrorAction SilentlyContinue |
            Where-Object { $_.Name -notlike '*.launch-info.json' }
        if ($matches -and $matches.Count -gt 0) {
            foreach ($m in $matches) {
                Move-TaskItem -FilePath $m.FullName -DestDir $doneDir
                # Clean up launch-info
                $infoFile = Join-Path $progressDir "$($m.BaseName).launch-info.json"
                if (Test-Path $infoFile) { Remove-Item $infoFile -Force }
            }
            Write-Host "Task moved to done." -ForegroundColor Green
        } else {
            Write-Error "No task matching '*$Arg1*' in in-progress."
        }
    }

    { $_ -in "claude-resume", "resume" } {
        if (-not $Arg1) { Write-Error "Usage: launch.ps1 $Action <session-id>"; return }
        $cfg = $ToolConfig[$Tool]
        $zone = Get-NextEmptyZone -MonitorFilter $Monitor
        $titleLabel = if ($DisplayName) { $DisplayName } else { 'resume ' + $Arg1 }
        switch ($cfg.ResumeStyle) {
            'flag'       { $resumeCmd = $cfg.Exe + ' --resume ' + $Arg1 }
            'subcommand' { $resumeCmd = $cfg.Exe + ' resume ' + $Arg1 }
            'flag-eq'    { $resumeCmd = $cfg.Exe + ' --resume=' + $Arg1 }
        }
        $cmdLine = 'title ' + $cfg.Label + ': ' + $titleLabel + ' && ' + $resumeCmd
        $p = Start-Process cmd -ArgumentList '/k', $cmdLine -WorkingDirectory $env:USERPROFILE -PassThru
        if ($zone -and $p) { Set-WindowZone -ProcessId $p.Id -Zone $zone }

        # Inject /rename + /color if DisplayName was provided
        if ($DisplayName -and $p -and $cfg.SupportsRename) {
            $launchTime = Get-Date
            $resolvedId = Set-InputBarName -DisplayName $DisplayName -Color $BarColor -ProcessId $p.Id -WorkDir $env:USERPROFILE -LaunchedAfter $launchTime
        }
    }

    { $_ -in "claude-new", "new" } {
        $cfg = $ToolConfig[$Tool]
        $dir = if ($Arg1) { $Arg1 } else { $env:USERPROFILE }
        $prompt = if ($Rest) { $Rest -join ' ' } else { $null }
        $zone = Get-NextEmptyZone -MonitorFilter $Monitor
        $leaf = Split-Path $dir -Leaf
        if ($prompt) {
            if ($Tool -eq 'claude') {
                $cmdLine = 'cd /d "' + $dir + '" && title ' + $cfg.Label + ': ' + $leaf + ' && ' + $cfg.Exe + ' --append-system-prompt "' + $prompt + '"'
            } else {
                switch ($cfg.PromptStyle) {
                    'positional'       { $promptArg = ' "' + $prompt + '"' }
                    'interactive-flag' { $promptArg = ' -i "' + $prompt + '"' }
                }
                $cmdLine = 'cd /d "' + $dir + '" && title ' + $cfg.Label + ': ' + $leaf + ' && ' + $cfg.Exe + $promptArg
            }
        } else {
            $cmdLine = 'cd /d "' + $dir + '" && title ' + $cfg.Label + ': ' + $leaf + ' && ' + $cfg.Exe
        }
        $p = Start-Process cmd -ArgumentList '/k', $cmdLine -WorkingDirectory $dir -PassThru
        if ($zone -and $p) { Set-WindowZone -ProcessId $p.Id -Zone $zone }
    }

    { $_ -in "claude-skip", "skip" } {
        $cfg = $ToolConfig[$Tool]
        $dir = if ($Arg1) { $Arg1 } else { $env:USERPROFILE }
        $prompt = if ($Rest) { $Rest -join ' ' } else { $null }
        $zone = Get-NextEmptyZone -MonitorFilter $Monitor
        $titleLabel = if ($DisplayName) { $DisplayName } else { Split-Path $dir -Leaf }
        if ($prompt) {
            if ($Tool -eq 'claude') {
                $cmdLine = 'cd /d "' + $dir + '" && title ' + $cfg.Label + ': ' + $titleLabel + ' && ' + $cfg.Exe + ' ' + $cfg.SkipPermsFlag + ' --append-system-prompt "' + $prompt + '"'
            } else {
                switch ($cfg.PromptStyle) {
                    'positional'       { $promptArg = ' "' + $prompt + '"' }
                    'interactive-flag' { $promptArg = ' -i "' + $prompt + '"' }
                }
                $cmdLine = 'cd /d "' + $dir + '" && title ' + $cfg.Label + ': ' + $titleLabel + ' && ' + $cfg.Exe + ' ' + $cfg.SkipPermsFlag + $promptArg
            }
        } else {
            $cmdLine = 'cd /d "' + $dir + '" && title ' + $cfg.Label + ': ' + $titleLabel + ' && ' + $cfg.Exe + ' ' + $cfg.SkipPermsFlag
        }
        $p = Start-Process cmd -ArgumentList '/k', $cmdLine -WorkingDirectory $dir -PassThru
        if ($zone -and $p) { Set-WindowZone -ProcessId $p.Id -Zone $zone }

        # Inject /rename + /color if DisplayName was provided (e.g. master session)
        if ($DisplayName -and $p -and $cfg.SupportsRename) {
            $launchTime = Get-Date
            $resolvedId = Set-InputBarName -DisplayName $DisplayName -Color $BarColor -ProcessId $p.Id -WorkDir $dir -LaunchedAfter $launchTime
        }
    }

    "refresh" {
        $envName = if ($Arg1) { $Arg1.ToLower() } else { "dev" }
        switch ($envName) {
            "dev"  { $path = "C:\Solutions\Python\LifeRunnerV2dev" }
            "prod" { $path = "C:\Solutions\Python\LifeRunnerV2prod" }
            default { Write-Error "Unknown env: $envName. Use 'dev' or 'prod'"; return }
        }
        $script = Join-Path $path "refresh.cmd"
        if (-not (Test-Path $script)) { Write-Error "Not found: $script"; return }
        Write-Host "Launching $script (will request admin elevation)..."
        Start-Process cmd -ArgumentList "/k", "cd /d `"$path`" && refresh.cmd" -WorkingDirectory $path -Verb RunAs
    }

    "screenshotter" {
        $python = "C:\miniconda3\envs\python312\python.exe"
        $grabber = "C:\Users\admin\.claude\skills\rad_screenshotter\grabber.py"
        $duration = if ($Arg1) { $Arg1 } else { "300" }
        $outDir = & $python -c "import os, datetime; d = os.path.join('C:/Documents/Screengrabs', datetime.datetime.now().strftime('%Y%m%d_%H%M%S')); os.makedirs(d, exist_ok=True); print(d)"
        Write-Host "Output: $outDir"
        Write-Host "Duration: $duration seconds"
        Start-Process cmd -ArgumentList "/k", "title Screenshotter && `"$python`" -u `"$grabber`" `"$outDir`" $duration" -WorkingDirectory $HOME
        Write-Host "Screenshotter launched. Use 'launch.ps1 screenshotter-stop' to end early."
    }

    "screenshotter-stop" {
        $pidFile = "C:\Users\admin\.claude\skills\rad_screenshotter\grabber.pid"
        if (Test-Path $pidFile) {
            $lines = Get-Content $pidFile
            $outDir = $lines[1]
            New-Item -Path (Join-Path $outDir "STOP") -ItemType File -Force | Out-Null
            Write-Host "Stop signal sent. Grabber will finish current cycle and exit."
            Write-Host "Output dir: $outDir"
        } else {
            Write-Error "No running screenshotter found (no PID file)"
        }
    }

    "cmd" {
        $command = @($Arg1) + $Rest -join " "
        if (-not $command) { Write-Error "Usage: launch.ps1 cmd <command>"; return }
        $zone = Get-NextEmptyZone -MonitorFilter $Monitor
        $p = Start-Process cmd -ArgumentList "/k", $command -PassThru
        if ($zone -and $p) { Set-WindowZone -ProcessId $p.Id -Zone $zone }
    }

    default {
        Write-Host @"

  launch.ps1 -- Universal launcher with zone tiling

  OPTIONS:
    -Tool <claude|codex|copilot>   CLI tool to use (default: claude)
    -Monitor <N>                   Target monitor (1=landscape, 2=portrait)
    -SkipPerms                     Skip permission prompts
    -Resume                        Resume previous session
    -Next                          Pick next task from backlog
    -Count <N>                     Number of tasks to launch
    -Remote                        Run on VM via SSH (default: ubuntu1)
    -RemoteHost <name>             SSH host for remote (default: ubuntu1)

  TASK LAUNCHER:
    task <slug>                  Launch session for a backlog task
    task <slug> -Tool codex      Launch with Codex CLI instead of Claude
    task <slug> -Tool copilot    Launch with Copilot CLI instead of Claude
    task <slug> -Resume          Resume previous session for that task
    task <slug> -SkipPerms       Launch with permissions skipped
    task <slug> -Remote          Run on ubuntu1 VM (auto skip-perms)
    task <slug> -Remote -Tool codex   Remote with Codex (--full-auto)
    task <slug> -Remote -RemoteHost ubuntu2   Run on ubuntu2 instead
    task -Next                   Pick next task from backlog
    task -Next -Count 3          Launch next 3 tasks in parallel
    status                       Show active zones and running tasks
    done <slug>                  Move task from in-progress to done

  SESSIONS (aliases: new, skip, resume):
    claude-new <dir> [prompt]    New interactive session
    claude-skip <dir> [prompt]   Session with perms skipped
    claude-resume <session-id>   Resume a session by ID

  SERVICES:
    refresh <dev|prod>           Restart LifeRunner (admin elevated)

  TOOLS:
    screenshotter [seconds]      Start screenshot capture (default 300s)
    screenshotter-stop           Stop running screenshotter
    cmd <command>                Run command in new tiled window

"@ -ForegroundColor Cyan
    }
}
